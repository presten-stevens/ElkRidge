         ```typescript
         async sendAttachment(
           phone: string,
           message: string,
           file: { buffer: Buffer; originalname: string; mimetype: string },
         ): Promise<{ guid: string }> {
           const tempGuid = crypto.randomUUID();
           const form = new FormData();
           form.append('chatGuid', `any;-;${phone}`);
           form.append('tempGuid', `temp-${tempGuid}`);
           form.append('message', message);
           form.append(
             'attachment',
             new Blob([file.buffer], { type: file.mimetype }),
             file.originalname,
           );

           const url = new URL('/api/v1/message/attachment', this.baseUrl);
           url.searchParams.set('password', this.password);

           let response: Response;
           try {
             response = await fetch(url.toString(), {
               method: 'POST',
               body: form,
               signal: AbortSignal.timeout(60_000),
             });
           } catch {
             throw new AppError(
               'BlueBubbles server is unreachable',
               ERROR_CODES.BB_OFFLINE,
               true,
               503,
             );
           }

           const body = (await response.json()) as { status: number; data: { guid: string }; error?: { message?: string } };
           if (body.status !== 200) {
             throw new AppError(
               body.error?.message ?? 'Attachment send failed',
               ERROR_CODES.ATTACHMENT_SEND_FAILED,
               false,
               502,
             );
           }
           return body.data;
         }
         ```
       - Add `downloadAttachment` method to `BlueBubblesClient`:
         ```typescript
         async downloadAttachment(guid: string): Promise<{
           body: ReadableStream<Uint8Array> | null;
           contentType: string;
           contentLength: string;
         }> {
           const url = new URL(
             `/api/v1/attachment/${encodeURIComponent(guid)}/download`,
             this.baseUrl,
           );
           url.searchParams.set('password', this.password);

           let response: Response;
           try {
             response = await fetch(url.toString(), {
               signal: AbortSignal.timeout(60_000),
             });
           } catch {
             throw new AppError(
               'BlueBubbles server is unreachable',
               ERROR_CODES.BB_OFFLINE,
               true,
               503,
             );
           }

           if (!response.ok) {
             throw new AppError(
               'Attachment not found',
               ERROR_CODES.ATTACHMENT_NOT_FOUND,
               false,
               404,
             );
           }

           return {
             body: response.body,
             contentType: response.headers.get('content-type') ?? 'application/octet-stream',
             contentLength: response.headers.get('content-length') ?? '',
           };
         }
         ```

       import { getBBClient } from '../services/bluebubbles.js';

       export const attachmentsRouter = Router();

       attachmentsRouter.get('/attachments/:guid', async (req, res) => {
         const client = getBBClient();
         const result = await client.downloadAttachment(req.params.guid);

         res.set('Content-Type', result.contentType);
         if (result.contentLength) {
           res.set('Content-Length', result.contentLength);
         }

         if (result.body) {
           Readable.fromWeb(result.body as any).pipe(res);
         } else {
           res.end();
         }
       });
