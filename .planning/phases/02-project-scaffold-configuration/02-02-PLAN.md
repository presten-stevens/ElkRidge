import { parsePhoneNumberFromString } from 'libphonenumber-js';

export function normalizePhone(input: string, defaultCountry: string = 'US'): string {
  const phone = parsePhoneNumberFromString(input, defaultCountry as any);
  if (!phone || !phone.isValid()) {
    throw new Error(`Invalid phone number: ${input}`);
  }
  return phone.number; // E.164 format, e.g., '+12135551234'
}

5. Create `src/middleware/__tests__/logger.test.ts`:

Test that Pino's redaction actually replaces password values. Strategy: Create a logger instance with a writable stream destination, log an object with a password field, capture the output, and assert the password value is replaced with "[REDACTED]".

```typescript
import { describe, it, expect } from 'vitest';
import pino from 'pino';

describe('logger redaction', () => {
  function createTestLogger() {
    const output: string[] = [];
    const dest = new Writable({
      write(chunk, _encoding, callback) {
        output.push(chunk.toString());
        callback();
      },
    });
    const testLogger = pino({
      redact: {
        paths: [
          'password',
          '*.password',
          '*.bluebubbles_password',
          '*.BLUEBUBBLES_PASSWORD',
          'req.headers.authorization',
          'req.query.password',
        ],
        censor: '[REDACTED]',
      },
    }, dest);
    return { logger: testLogger, output };
  }

  it('redacts top-level password field', () => {
    const { logger, output } = createTestLogger();
    logger.info({ password: 'my-secret-pw' }, 'test');
    logger.flush();
    const logLine = output[0];
    expect(logLine).toContain('[REDACTED]');
    expect(logLine).not.toContain('my-secret-pw');
  });

  it('redacts nested bluebubbles_password field', () => {
    const { logger, output } = createTestLogger();
    logger.info({ config: { bluebubbles_password: 'bb-secret' } }, 'test');
    logger.flush();
    const logLine = output[0];
    expect(logLine).toContain('[REDACTED]');
    expect(logLine).not.toContain('bb-secret');
  });
});
```

NOTE: Import `Writable` from `node:stream` at top of test file.

6. Create `src/middleware/__tests__/error-handler.test.ts` (SECR-04 -- error handler must never leak credentials):

Test that the errorHandler middleware never includes the BLUEBUBBLES_PASSWORD value in its JSON response body, even when the error message itself contains the password.

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { errorHandler } from '../../middleware/error-handler.js';
import type { Request, Response, NextFunction } from 'express';

describe('errorHandler credential safety (SECR-04)', () => {
  const FAKE_BB_PASSWORD = 'super-secret-bb-password-12345';
  let mockReq: Partial<Request>;
  let mockRes: Partial<Response>;
  let mockNext: NextFunction;
  let jsonBody: any;

  beforeEach(() => {
    mockReq = { method: 'GET', url: '/test' } as Partial<Request>;
    jsonBody = null;
    mockRes = {
      status: vi.fn().mockReturnThis(),
      json: vi.fn((body) => { jsonBody = body; }),
    } as unknown as Partial<Response>;
    mockNext = vi.fn();
  });

  it('response body does not contain BLUEBUBBLES_PASSWORD value', () => {
    const error = new Error('Something went wrong');
    errorHandler(error, mockReq as Request, mockRes as Response, mockNext);
    const responseStr = JSON.stringify(jsonBody);
    expect(responseStr).not.toContain(FAKE_BB_PASSWORD);
  });

  it('returns structured { error: { message, code } } JSON', () => {
    const error = new Error('Bad request');
    errorHandler(error, mockReq as Request, mockRes as Response, mockNext);
    expect(jsonBody).toHaveProperty('error');
    expect(jsonBody.error).toHaveProperty('message');
    expect(jsonBody.error).toHaveProperty('code');
  });

  it('does not leak password even when error.message contains it', () => {
    // Simulate a scenario where the BB password accidentally ends up in an error message
    const error = new Error(`Connection failed with password=${FAKE_BB_PASSWORD}`);
    errorHandler(error, mockReq as Request, mockRes as Response, mockNext);
    const responseStr = JSON.stringify(jsonBody);
    // The error handler should sanitize or genericize the message.
    // At minimum, the raw password must not appear in the response.
    // NOTE: If this test fails, the error handler needs to sanitize err.message
    // before including it in the response body. The implementation should strip
    // or replace any occurrence of env.BLUEBUBBLES_PASSWORD from the message,
    // or use a generic message for 500 errors.
    expect(responseStr).not.toContain(FAKE_BB_PASSWORD);
  });
});
```

IMPORTANT: Test 3 above ("does not leak password even when error.message contains it") may require updating the error handler implementation in Plan 01's `src/middleware/error-handler.ts` to sanitize `err.message` before including it in the JSON response. The current implementation passes `err.message` directly. Two approaches to fix:

Option A (preferred -- simpler): For 500 errors, always return a generic "Internal server error" message instead of `err.message`. Only expose `err.message` for 4xx client errors (where the message is intentionally set by route handlers).

Option B: Import `env.BLUEBUBBLES_PASSWORD` and do `err.message.replaceAll(env.BLUEBUBBLES_PASSWORD, '[REDACTED]')` before including in response.

The executor should implement Option A: only expose err.message when status < 500. For status >= 500, return "Internal server error". This is both more secure (defense in depth) and simpler. Update src/middleware/error-handler.ts accordingly during this task if needed.
