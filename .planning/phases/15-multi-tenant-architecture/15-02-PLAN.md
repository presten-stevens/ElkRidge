export function authMiddleware(req, _res, next) {
  if (!env.API_KEY) { next(); return; }
  const authHeader = req.headers.authorization;
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    next(new AppError('Invalid or missing API key', ERROR_CODES.AUTH_FAILURE, false, 401));
    return;
  }
  const token = authHeader.slice(7);
  if (token !== env.API_KEY) { next(new AppError(...)); return; }
  next();
}
```

       // Manager env MUST be imported first -- fail fast on invalid config
       import { managerEnv } from './config/env.js';
       import { createManagerApp } from './app.js';
       import { connectPM2, disconnectPM2 } from './services/pm2-manager.js';
       import { logger } from '../middleware/logger.js';

       const app = createManagerApp();

       async function start() {
         try {
           await connectPM2();
           logger.info('Connected to PM2 daemon');
         } catch (err) {
           logger.error({ err: err instanceof Error ? err.message : String(err) }, 'Failed to connect to PM2 daemon');
           process.exit(1);

         app.listen(managerEnv.MANAGER_PORT, '127.0.0.1', () => {
           logger.info({ port: managerEnv.MANAGER_PORT }, 'Management API started');
         });
       }

       function shutdown() {
         logger.info('Shutting down management API');
         disconnectPM2();
         process.exit(0);
       }

       process.on('SIGTERM', shutdown);
       process.on('SIGINT', shutdown);

       start();
       ```
